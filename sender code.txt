#include <esp_now.h>
#include <WiFi.h>
#include <DHT.h>

// --- PIN DEFINITIONS ---
#define DHTPIN 4
#define RAIN_PIN 34
#define SOIL_DIGITAL 32

#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// --- RECEIVER MAC ADDRESS ---
// --- RECEIVER MAC ADDRESS ---
uint8_t receiverMAC[] = {0xD0, 0xEF, 0x76, 0x44, 0x8E, 0xD0};

// --- DATA STRUCTURE ---
typedef struct struct_message {
    float temp;
    float hum;
    int soil;
    int rain;
} struct_message;

struct_message myData;
esp_now_peer_info_t peerInfo;

// Callback for transmission status
void OnDataSent(const wifi_tx_info_t *info, esp_now_send_status_t status) {
  Serial.print("Transmission Status: ");
  Serial.println(status == ESP_NOW_SEND_SUCCESS ? "SUCCESS (Ack Received)" : "FAIL (No Ack)");

  // Optional: print MAC from tx info
  if (info) {
    Serial.print("To MAC: ");
    for (int i = 0; i < 6; i++) {
      Serial.printf("%02X", info->des_addr[i]);
      if (i < 5) Serial.print(":");
    }
    Serial.println();
  }
}
void setup() {
  Serial.begin(115200);
  dht.begin();
  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  // ✅ FIXED (no cast needed)
  esp_now_register_send_cb(OnDataSent);
  
  memset(&peerInfo, 0, sizeof(peerInfo)); 
  memcpy(peerInfo.peer_addr, receiverMAC, 6);
  peerInfo.channel = 0;  
  peerInfo.encrypt = false;
  
  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }

  analogReadResolution(12);
  pinMode(SOIL_DIGITAL, INPUT);
  
  Serial.println("AI Field Node: Initialized");
}

void loop() {
  // 1. Capture Sensor Data
  myData.temp = dht.readTemperature();
  myData.hum = dht.readHumidity();
  myData.soil = digitalRead(SOIL_DIGITAL); 
  
  int rainRaw = analogRead(RAIN_PIN);
  myData.rain = map(rainRaw, 4095, 1000, 0, 100);
  myData.rain = constrain(myData.rain, 0, 100);

  // --- SERIAL MONITOR OUTPUT ---
// --- SERIAL MONITOR OUTPUT ---
Serial.println("\n--- CAPTURING FIELD DATA ---");

Serial.print("Temperature: ");
Serial.print(myData.temp);
Serial.println(" °C");

Serial.print("Humidity:    ");
Serial.print(myData.hum);
Serial.println(" %");

Serial.print("Soil Status: ");
if (myData.soil == 1) {
  Serial.println("DRY");
} else {
  Serial.println("WET");
}

Serial.print("Rain Level:  ");
Serial.print(myData.rain);
Serial.println(" %");

// Transmission status will print from callback
Serial.println("----------------------------");

  // 3. Send Data
  esp_now_send(receiverMAC, (uint8_t *) &myData, sizeof(myData));

  delay(7000);
}
